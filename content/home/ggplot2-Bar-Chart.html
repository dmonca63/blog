dft<-Teams%>%
  filter(yearID==1980)%>%
  select(name,HR)%>%
  arrange(HR)
## Warning: package 'bindrcpp' was built under R version 3.4.2
ggplot()+
  geom_bar(data = dft, aes(x=name,y=HR),stat = "identity",color="white", fill="pink")+
  coord_flip()
